<objective>
Dashboard-side remote control: toggle button, transparent overlay for event capture, coordinate mapping, and WS message dispatch.

Purpose: This is the user-facing half of remote control. When the user activates the toggle, a transparent overlay captures their mouse clicks, keyboard input, and scroll gestures on the preview, translates coordinates from preview-space to real-browser-space using `previewScale`, and sends them as WS messages to the extension (which dispatches them via CDP per Plan 01).

Output: dashboard.html has a toggle button and overlay div; dashboard.js has event capture/forwarding logic; dashboard.css has overlay and active-state styling.
</objective>

6. Add the event capture IIFE for remote control (place after the PiP drag IIFE, around line 1892):

```javascript
// Remote control event forwarding
(function initRemoteControl() {
  if (!remoteOverlay) return;

  // --- CLICK forwarding (CONTROL-01, per D-01, D-02, D-03) ---
  remoteOverlay.addEventListener('mousedown', function (e) {
    if (!remoteControlOn || !ws || ws.readyState !== WebSocket.OPEN) return;
    e.preventDefault();
    e.stopPropagation();

    // Get click position relative to the preview container
    var rect = remoteOverlay.getBoundingClientRect();
    var clickX = e.clientX - rect.left;
    var clickY = e.clientY - rect.top;

    // Reverse-scale to real browser coordinates (per D-01)
    var realX = Math.round(clickX / previewScale);
    var realY = Math.round(clickY / previewScale);

    // Build modifiers bitmask (Claude's discretion: forward modifiers)
    var modifiers = 0;
    if (e.altKey) modifiers |= 1;
    if (e.ctrlKey) modifiers |= 2;
    if (e.metaKey) modifiers |= 4;
    if (e.shiftKey) modifiers |= 8;

    ws.send(JSON.stringify({
      type: 'dash:remote-click',
      payload: { x: realX, y: realY, button: 'left', modifiers: modifiers },
      ts: Date.now()
    }));
  });

  // --- KEYBOARD forwarding (CONTROL-02, per D-04, D-05, D-06) ---
  // Keyboard events must be captured at document level when remote control is active
  // because the overlay div does not naturally receive keyboard focus.
  document.addEventListener('keydown', function (e) {
    if (!remoteControlOn || !ws || ws.readyState !== WebSocket.OPEN) return;
    // Do not capture if user is typing in a dashboard input field
    var activeTag = document.activeElement ? document.activeElement.tagName : '';
    if (activeTag === 'INPUT' || activeTag === 'TEXTAREA' || activeTag === 'SELECT') return;

    e.preventDefault();

    var modifiers = 0;
    if (e.altKey) modifiers |= 1;
    if (e.ctrlKey) modifiers |= 2;
    if (e.metaKey) modifiers |= 4;
    if (e.shiftKey) modifiers |= 8;

    ws.send(JSON.stringify({
      type: 'dash:remote-key',
      payload: {
        type: 'keyDown',
        key: e.key,
        code: e.code,
        text: e.key.length === 1 ? e.key : '',
        modifiers: modifiers
      },
      ts: Date.now()
    }));

    // For printable characters, also send a 'char' event for text input
    if (e.key.length === 1) {
      ws.send(JSON.stringify({
        type: 'dash:remote-key',
        payload: {
          type: 'char',
          key: e.key,
          code: e.code,
          text: e.key,
          modifiers: modifiers
        },
        ts: Date.now()
      }));
    }
  });

  document.addEventListener('keyup', function (e) {
    if (!remoteControlOn || !ws || ws.readyState !== WebSocket.OPEN) return;
    var activeTag = document.activeElement ? document.activeElement.tagName : '';
    if (activeTag === 'INPUT' || activeTag === 'TEXTAREA' || activeTag === 'SELECT') return;

    e.preventDefault();

    var modifiers = 0;
    if (e.altKey) modifiers |= 1;
    if (e.ctrlKey) modifiers |= 2;
    if (e.metaKey) modifiers |= 4;
    if (e.shiftKey) modifiers |= 8;

    ws.send(JSON.stringify({
      type: 'dash:remote-key',
      payload: {
        type: 'keyUp',
        key: e.key,
        code: e.code,
        text: '',
        modifiers: modifiers
      },
      ts: Date.now()
    }));
  });

  // --- SCROLL forwarding (CONTROL-03, per D-07, D-08) ---
  // Throttle scroll events to ~60fps (16ms) to avoid overwhelming the WS connection
  var scrollThrottleTimer = null;
  remoteOverlay.addEventListener('wheel', function (e) {
    if (!remoteControlOn || !ws || ws.readyState !== WebSocket.OPEN) return;
    e.preventDefault();
    e.stopPropagation();

    if (scrollThrottleTimer) return; // Throttled
    scrollThrottleTimer = setTimeout(function () { scrollThrottleTimer = null; }, 16);

    // Get scroll position relative to the preview
    var rect = remoteOverlay.getBoundingClientRect();
    var scrollX = e.clientX - rect.left;
    var scrollY = e.clientY - rect.top;

    // Reverse-scale coordinates (per D-08)
    var realX = Math.round(scrollX / previewScale);
    var realY = Math.round(scrollY / previewScale);

    ws.send(JSON.stringify({
      type: 'dash:remote-scroll',
      payload: { x: realX, y: realY, deltaX: Math.round(e.deltaX), deltaY: Math.round(e.deltaY) },
      ts: Date.now()
    }));
  }, { passive: false }); // passive: false required to call preventDefault on wheel
})();
```

Key design decisions made (Claude's discretion):
- Modifier keys (Ctrl, Alt, Shift, Meta) ARE forwarded with click/key events -- more useful than not
- Scroll is throttled at ~60fps (16ms interval) -- matches display refresh
- Blue border (#3b82f6) indicates remote control active -- distinct from orange automation border
- No cursor indicator dot at last click position -- keeping it simple for v1
- Only left-click forwarded (right-click and double-click deferred to future)
